## 열거형 -  iota

상수를 사용하는 일정한 규칙에 따라 숫자를 계산 및 증가시키는 묶음

```go
package main

import "fmt"

func main(){
	//iota를 이용한 상수 선언
	const(
		A = iota //0
		B //1
		C //2
	)

	
	//iota의 증가하는 특성을 이용한 수식을 만들 수 있음
	const(
		Jan = iota + 1 //0+1
		Feb //1+1
		Mar //2+1
		Apr //3+1
	)

	// '_'를 이용한 상수 선언 제외
	const(
		_ = iota * 2 //0 제외
		X //2
		Y //4
		_ //6 제외
		Z //8 
	)

	fmt.Println(A,B,C)
	fmt.Println(Jan, Feb, Mar, Apr)
	fmt.Println(X,Y,Z)
	

}

>> 0 1 2
>> 1 2 3 4
>> 2 4 8
```
